/**
 * @function
 * @description html
 * @param  {string} argHtml 
 * @returns {string}
 * @example
 * // returns '&lt;p&gt;Hello, World!&lt;/p&gt;'
 * encodeHtml('<p>Hello, World!</p>')
 */
export const encodeHtml = (argHtml: string): string => {
  if (!argHtml || argHtml.length === 0) {
    return ''
  }
  const htmlEntities: StringObject = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    "'": '&apos;',
    '"': '&quot;',
    ' ': '&nbsp;',
    '\n': '<br>',
  }
  return argHtml.replace(/[&<>"' \n]/g, function (char) {
    return htmlEntities[char]
  })
}
 
/**
 * @function decodeHtml
 * @description HTML
 * @param {string} argHtml - 
 * @returns {string} 
 * @example
 * //  "&lt;p&gt;Hello, World!&lt;/p&gt;"
 * decodeHtml("<p>Hello, World!</p>");
 */
export const decodeHtml = (argHtml: string): string => {
  const htmlEntities: StringObject = {
    quot: '"',
    apos: "'",
    lt: '<',
    gt: '>',
    nbsp: ' ',
    amp: '&',
  }
  return argHtml.replace(/&([^;]+);/g, (match, entity) => {
    return htmlEntities[entity] || match
  })
}

/**
 * @function formatTime
 * @description 
 * @param {string|number|Date} date - 
 * @param {string} fmt - 'YYYY-MM-DD HH:mm:ss'EEEE:qS
 * @param {string} emptyTip - datefalse''
 * @returns {string} 
 * @example
 * //  '2022-01-01 00:00:00'
 * formatTime(new Date(2022, 0, 1));
 */
export const formatTime = (
  date: string | number | Date = +new Date(),
  fmt: string = 'YYYY-MM-DD HH:mm:ss',
  emptyTip: string = ''
): string => {
  if (!date && date !== 0) {
    return emptyTip
  }
  if (isNaN(+date)) {
    return emptyTip
  }
  if (typeof date === 'number') {
    if (date.toString().length === 10) {
      date = date * 1000
    }
  }
  date = new Date(+date)
  const fmtMap: { [key: string]: number } = {
    'M+': date.getMonth() + 1,
    'D+': date.getDate(),
    'h+': date.getHours() % 12 === 0 ? 12 : date.getHours() % 12,
    'H+': date.getHours(),
    'm+': date.getMinutes(),
    's+': date.getSeconds(),
    'q+': Math.floor((date.getMonth() + 3) / 3),
    S: date.getMilliseconds(),
  }
  const week = {
    0: '\u65e5',
    1: '\u4e00',
    2: '\u4e8c',
    3: '\u4e09',
    4: '\u56db',
    5: '\u4e94',
    6: '\u516d',
  }
  const temY = fmt.match(/(Y+)/)
  if (temY) {
    fmt = fmt.replace(
      temY[0],
      (date.getFullYear() + '').substring(4 - temY[0].length)
    )
  }
  const temWeek = fmt.match(/(E+)/)
  if (temWeek) {
    const temWeekName = ['', '\u5468']
    fmt = fmt.replace(
      temWeek[0],
      (temWeekName[temWeek[0].length] || '\u661f\u671f') +
        safeData(week, date.getDay().toString(), '')
    )
  }
  for (let k in fmtMap) {
    let temMatch = fmt.match(new RegExp('(' + k + ')'))
    if (temMatch) {
      fmt = fmt.replace(
        temMatch[0],
        temMatch[0].length === 1
          ? fmtMap[k].toString()
          : ('00' + fmtMap[k].toString()).substring(('' + fmtMap[k]).length)
      )
    }
  }
  return fmt
}

export const friendlyTime = (

  fmt: string = 'YYYY-MM-DD HH:mm:ss',
  emptyTip: string = ''
): string => {
  if (!date && date !== 0) {
    return emptyTip
  }
  if (isNaN(+date)) {
    return emptyTip
  }
  if (typeof date === 'number' || typeof date === 'string') {
    if (date.toString().length === 10) {
      date = date.toString() + '000'
    }
  }
  date = new Date(+date)
  let diff = (new Date().getTime() - date.getTime()) / 1000
  let dayDiff = Math.floor(diff / 86400)
  let isValidDate =
    Object.prototype.toString.call(date) === '[object Date]' &&
    !isNaN(date.getTime())
  if (!isValidDate) {
    console.warn('friendlyTime:', 'not a valid date')
    return emptyTip || '--'
  }
  if (isNaN(dayDiff) || dayDiff < 0 || dayDiff >= 365) {
    return formatTime(date, fmt)
  }
  return (
    (dayDiff === 0 &&
      ((diff < 60 && '') ||
        (diff < 120 && '1') ||
        (diff < 3600 && Math.floor(diff / 60) + '') ||
        (diff < 7200 && '1') ||
        (diff < 86400 && Math.floor(diff / 3600) + ''))) ||
    (dayDiff === 1 && '') ||
    (dayDiff < 7 && dayDiff + '') ||
    (dayDiff < 31 && Math.ceil(dayDiff / 7) + '') ||
    (dayDiff < 365 && Math.ceil(dayDiff / 31) + '')
  ).toString()
}

/**
 * @function formatSize
 * @description byte
 * @param {number} argData - byte
 * @param {number} argNum - 2
 * @param {number} argIndex - : 0:b 1:k 2:m0
 * @param {number} argRate - 1024
 * @param {string[]} unit - ['B', 'K', 'M', 'G', 'T', 'P']
 * @returns {string} 
 * @example
 * // returns '1.95K'
 * formatSize(2000, 2, 0, 1024, ['B', 'K', 'M', 'G', 'T', 'P'])
 */
export const formatSize = (
  argData: number,
  argNum: number = 2,
  argIndex: number = 0,
  argRate: number = 1024,
  unit: string[] = ['B', 'K', 'M', 'G', 'T', 'P']
): string => {
  let list = unit
  if (!argData) {
    return '0' + list[argIndex]
  }
  let len = list.length
  let nowIndex = len - 1
  let temData: number = 0
  for (let i = 0; i < len; i++) {
    temData = argData / Math.pow(argRate, i)
    if (temData < argRate) {
      nowIndex = argIndex + i
      break
    }
  }
  return +toFixed(temData, argNum) + list[nowIndex]
}
/**
 * @function formatNumber
 * @description k/w
 * @param {number} argData - 
 * @param {number} argNum - 2
 * @returns {string} 
 * @example
 * // returns '2k'
 * formatNumber(2000, 2)
 */
export const formatNumber = (argData: number, argNum: number = 2): string => {
  if (argData < 1000) {
    return argData.toString()
  } else if (argData < 10000) {
    return +toFixed(argData / 1000, argNum) + 'k'
  } else {
    return +toFixed(argData / 10000, argNum) + 'w'
  }
}

  let radix = chars.length

/**

 * @description toFixed
 * @date 2020-03-01
 * @param {string|number} argData - 
 * @param {number} argNum - 2
 * @param {string} argType - round:,floor:,ceil:,abs:
 * @returns {string} - 
 * @example
 * // 
 * const data = 123.456;
 * const fixedData = toFixed(data, 2, 'round');
 * console.log(fixedData); // : '123.46'
 */
export const toFixed = (
  argData: string | number,
  argNum: number = 2,
  argType: 'round' | 'floor' | 'ceil' | 'abs' = 'round'
): string => {
  if (isNaN(+argData)) {
    return ''
  }
  const data =
    Math[argType](+argData * Math.pow(10, argNum)) / Math.pow(10, argNum)
  return data.toFixed(argNum)
}

/**
 * @function
 * @description key,
 * @param {any[]} list - 
 * @param {string} key - 
 * @param {string|object} extraData - string,object
 * @return {Object} - 
 * @example
 * // 
 * const list = [{ id: 1, name: 'Alice' }, { id: 2, name: 'Bob' }];
 * const obj = arrayToObj(list, 'id');
 * console.log(obj); // : { '1': { id: 1, name: 'Alice' }, '2': { id: 2, name: 'Bob' } }
 */
export const arrayToObj = (
  list: any[],
  key: string = '',
  extraData: string | object = ''
): AnyObject => {
  let obj: AnyObject = {}
  list.forEach((item: any, k) => {
    if (extraData) {
      if (typeof extraData === 'string') {
        obj[item[key] || key + k] = item[extraData] || ''
      } else if (typeof extraData === 'object') {
        obj[item[key] || key + k] = Object.assign(item, deepCopy(extraData))
      }
    } else {
      obj[item[key] || key + k] = item
    }
  })
  return obj
}
/**
 * @function 
 * @description 
 * @param {object} obj - 
 * @param {string} [key='key'] - key
 * @param {string} [value='value'] - value
 * @param {boolean} [isReverse=false] - true,false
 * @return {array} - 
 * @example 
// Example usage
const obj = { a: 1, b: 2 };
const result = objToArray(obj, 'id', 'v', true);
console.log(result); // Output: [{ v: 'a', id: 1 }, { v: 'b', id: 2 }]
*/
export const objToArray = (
  obj: { [key: string]: any },
  key: string = 'key',
  value: string = 'value',
  isReverse: boolean = false
): any[] => {
  const result: any[] = []
  for (const k in obj) {
    if (obj.hasOwnProperty(k)) {
      result.push({
        [key]: isReverse ? obj[k] : k,
        [value]: isReverse ? k : obj[k],
      })
    }
  }
  return result
}
/**

 * @param {Object} obj -

 * //

export const objToObj = (
  obj: { [key: string]: any },
  tmplObj: { [key: string]: any },
  isAssign: boolean = false

  for (const key in newObj) {
    const temKey: string = String(newObj[key] || key)
    if (typeof newObj[key] === 'object') {
